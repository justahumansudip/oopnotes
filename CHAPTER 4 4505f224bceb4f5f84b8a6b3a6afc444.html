Fundamental of Operator Overloading

Q: What is operator overloading?

A: Operator overloading is a feature in C++ that allows operators such as +, -, *, /, etc. to have different meanings when applied to user-defined data types.

Q: How is operator overloading done?

A: Operator overloading is done by defining a special function called operator function.

Program example:

#include<iostream>
using namespace std;
class Complex {
   private:
      int real, imag;
   public:
      Complex(int r = 0, int i =0) {
         real = r;   imag = i;
      }
      Complex operator + (Complex const &obj) {
         Complex res;
         res.real = real + obj.real;
         res.imag = imag + obj.imag;
         return res;
      }
      void print() {
         cout << real << " + i" << imag << endl;
      }
};
int main() {
   Complex c1(10, 5), c2(2, 4);
   Complex c3 = c1 + c2;
   c3.print();
}

Restriction on Operator Overloading

Q: What are the restrictions on operator overloading?

A: The restrictions on operator overloading are:

  • The overloaded operator must have at least one operand that is of user-defined data type.
  • The overloaded operator cannot change the number of operands.
  • The overloaded operator cannot change the precedence of the operator.

Program example:

#include<iostream>
using namespace std;
class Count {
   private:
      int value;
   public:
      Count(int v = 0) { value = v; }
      void operator ++ () { ++value; }
      void display() { cout<< "Count: " << value << endl; }
};
int main() {
   Count c1(5);
   ++c1;
   c1.display();
   ++(++c1);
   c1.display();
   return 0;
}

Operator Functions as a Class Members

Q: What are operator functions as class members?

A: Operator functions as class members are the functions that are declared inside the class and are used to overload operators.

Program example:

#include<iostream>
using namespace std;
class Distance {
   private:
      int feet;
      int inches;
   public:
      Distance() {
         feet = 0;
         inches = 0;
      }
      Distance(int f, int i) {
         feet = f;
         inches = i;
      }
      void operator = (const Distance &d) {
         feet = d.feet;
         inches = d.inches;
      }
      void display() {
         cout << "Feet: " << feet << ", Inches: " << inches << endl;
      }
};
int main() {
   Distance d1(11, 10), d2(5, 11);
   d1.display();
   d2.display();
   d1 = d2;
   d1.display();
   return 0;
}

Overloading Unary and Binary Operator

Q: What is the difference between unary and binary operators?

A: Unary operators are operators that operate on a single operand while binary operators are operators that operate on two operands.

Program example:

#include<iostream>
using namespace std;
class Complex {
   private:
      int real, imag;
   public:
      Complex(int r = 0, int i =0) {
         real = r;   imag = i;
      }
      Complex operator - () {
         Complex temp;
         temp.real = -real;
         temp.imag = -imag;
         return temp;
      }
      Complex operator + (Complex const &obj) {
         Complex res;
         res.real = real + obj.real;
         res.imag = imag + obj.imag;
         return res;
      }
      void print() {
         cout << real << " + i" << imag << endl;
      }
};
int main() {
   Complex c1(10, 5), c2(2, 4);
   Complex c3 = -c1;
   c3.print();
   Complex c4 = c1 + c2;
   c4.print();
   return 0;
}

Data Conversion

Q: What is data conversion?

A: Data conversion is the process of converting one data type to another.

Q: What are the types of data conversion?

A: The types of data conversion are:

  • Basic to Basic
  • Basic to User-defined
  • User-defined to Basic
  • User-defined to User-defined

Program example:

#include<iostream>
using namespace std;
class Distance {
   private:
      int feet;
      int inches;
   public:
      Distance() {
         feet = 0;
         inches = 0;
      }
      Distance(int f, int i) {
         feet = f;
         inches = i;
      }
      operator int() {
         return feet;
      }
};
int main() {
   Distance d1(11, 10);
   int feet = d1;
   cout << "Feet: " << feet << endl;
   return 0;
}

Basic to Basic

#include<iostream>
using namespace std;
int main(){
    int x = 10;
    double y = x; // basic to basic conversion
    cout << "x: " << x << endl;
    cout << "y: " << y << endl;
    return 0;
}

Basic to User-defined

#include<iostream>
using namespace std;
class Distance {
   private:
      int feet;
      int inches;
   public:
      Distance() {
         feet = 0;
         inches = 0;
      }
      Distance(int f, int i) {
         feet = f;
         inches = i;
      }
      void display() {
         cout << "Feet: " << feet << ", Inches: " << inches << endl;
      }
      operator double() {
         double total = feet + (double)inches/12;
         return total;
      }
};
int main() {
   Distance d1(11, 10);
   double total = d1; // basic to user-defined conversion
   cout << "Total feet: " << total << endl;
   return 0;
}

User-defined to User-defined

#include<iostream>
using namespace std;
class Distance {
   private:
      int feet;
      int inches;
   public:
      Distance() {
         feet = 0;
         inches = 0;
      }
      Distance(int f, int i) {
         feet = f;
         inches = i;
      }
      void display() {
         cout << "Feet: " << feet << ", Inches: " << inches << endl;
      }
      operator int() {
         int total = feet*12 + inches;
         return total;
      }
};
class Area {
   private:
      Distance d1;
      Distance d2;
   public:
      Area() {
         d1 = Distance(0, 0);
         d2 = Distance(0, 0);
      }
      Area(Distance x, Distance y) {
         d1 = x;
         d2 = y;
      }
      void display() {
         cout << "Distance 1: ";
         d1.display();
         cout << "Distance 2: ";
         d2.display();
         int total = d1 + d2; // user-defined to user-defined conversion
         cout << "Total inches: " << total << endl;
      }
};
int main() {
   Distance d1(11, 10), d2(5, 11);
   Area a1(d1, d2);
   a1.display();
   return 0;
}
